```splitfile
FROM demo/weather IMPORT rdu AS source_data

SQL CREATE TABLE monthly_summary AS ( \\
    SELECT to_char(date, 'YYYYMM') AS month, \\
        AVG(precipitation) AS average_precipitation, \\
        AVG(snowfall) AS average_snowfall \\
    FROM source_data \\
    GROUP BY month \\
    ORDER BY month ASC)
```
