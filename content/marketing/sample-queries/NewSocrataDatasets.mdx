```sql
-- Run a JOIN on two daily snapshots of the same dataset to see daily changes.
-- splitgraph/socrata is a collection of government datasets that we index daily.
-- This query finds new datasets on the data.texas.gov data portal.
SELECT
    new.domain AS domain,
    new.id AS id,
    new.name AS name
FROM
    "splitgraph/socrata:20200809".datasets old
RIGHT OUTER JOIN
    "splitgraph/socrata:20200810".datasets new
ON
    old.id = new.id
WHERE
    old.id IS NOT DISTINCT FROM NULL AND new.domain = 'data.texas.gov'
```
