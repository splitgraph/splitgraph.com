```sql
-- This query compares two snapshots of a catalog of all datasets on government open data portals
-- that Splitgraph indexes daily. It runs a JOIN to find new datasets on the data.texas.gov portal.
SELECT
    new.domain AS domain,
    new.id AS id,
    new.name AS name
FROM
    "splitgraph/socrata:20200809".datasets old
RIGHT OUTER JOIN
    "splitgraph/socrata:20200810".datasets new
ON
    old.id = new.id
WHERE
    old.id IS NOT DISTINCT FROM NULL AND new.domain = 'data.texas.gov'
```
